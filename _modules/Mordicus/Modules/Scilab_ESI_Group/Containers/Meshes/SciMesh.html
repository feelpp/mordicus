[docs]    def AllElementsIterator(self):
        class iterator:
            def __init__(self, simplices):
                self.simplices = simplices

            def __iter__(self):
                for simplex in self.simplices:
                    yield np.array(simplex)

        res = iterator(self.GetInternalStorage().get("nodes"))
        return res


    def __str__(self):
        return "Scilab mesh"
